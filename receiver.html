function connectWS() { let ws = new WebSocket("wss://your-server-url"); ws.onopen = () => console.log("Connected"); ws.onmessage = (event) => { if (event.data === "ALERT") { document.body.style.backgroundColor = "green"; setTimeout(() => { document.body.style.backgroundColor = "white"; }, 3000); } }; ws.onclose = () => { console.log("Disconnected, retrying in 2s..."); setTimeout(connectWS, 2000); // Προσπάθεια reconnect κάθε 2 δευτερόλεπτα }; ws.onerror = (err) => { console.error("WebSocket error:", err); ws.close(); }; } // Εκκίνηση WebSocket connectWS();
